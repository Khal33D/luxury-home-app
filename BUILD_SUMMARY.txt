SHOPIFY INTEGRATION EXAMPLE
================================================================================

// Hook usage in components:
import { useProducts, useCart } from '@/hooks/useShopify';

export function Example() {
  const { products, isLoading } = useProducts(20);
  const { cart, addToCart } = useCart();

  const handleAdd = (variantId: string) => {
    addToCart(variantId, 1)
      .then(cart => console.log('Added to cart:', cart))
      .catch(err => console.error(err));
  };

  return (
    <div>
      {products.map(product => (
        <button 
          key={product.id}
          onClick={() => handleAdd(product.variants[0].id)}
        >
          Add {product.title} - {formatPrice(product.priceRange.minVariantPrice.amount)}
        </button>
      ))}
    </div>
  );
}
